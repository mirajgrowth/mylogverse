navigator.geolocation.getCurrentPosition(pos => { let lat = pos.coords.latitude; let lon = pos.coords.longitude; fetch(`https://api.aladhan.com/v1/timings?latitude=${lat}&longitude=${lon}&method=2`) .then(res => res.json()) .then(data => { let t = data.data.timings; document.getElementById("times").innerHTML = `
Fajr: ${t.Fajr}
Zuhr: ${t.Dhuhr}
Asr: ${t.Asr}
Maghrib: ${t.Maghrib}
Isha: ${t.Isha}
`; }); });
